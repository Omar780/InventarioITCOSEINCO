function parseCSV(text){
const lines=text.split(/\r?\n/).filter(l=>l.trim());
if(lines.length<2){toast('Archivo vacío o sin datos','error');return;}
const sep=lines[0].includes(';')?';':',';
const headers=lines[0].split(sep).map(h=>h.trim().toUpperCase().replace(/^"|"$/g,''));
const rows=[];
for(let i=1;i<lines.length;i++){
const cells=splitLine(lines[i],sep);
if(cells.every(c=>!c.trim()))continue;
const obj={};
headers.forEach((h,idx)=>{obj[h]=(cells[idx]||'').trim().replace(/^"|"$/g,'');});
if(!obj['SISTEMA']&&obj['SISTEMA OPERATIVO']) obj['SISTEMA']=obj['SISTEMA OPERATIVO'];
obj._id='eq_'+Date.now()+'_'+i+'_'+Math.random().toString(36).slice(2,5);
rows.push(obj);
}
state.importBuf=rows;
const previewCols=['ITEM','AREA','USUARIO','TIPO','HOSTNAME','MARCA','MODELO','SERIE'];
document.getElementById('preview-count').textContent=rows.length+' registros encontrados';
document.getElementById('preview-head').innerHTML='<tr>'+previewCols.map(c=>`<th>${c}</th>`).join('')+'</tr>';
document.getElementById('preview-body').innerHTML=rows.slice(0,10).map(r=>'<tr>'+previewCols.map(c=>`<td>${esc(r[c]||'')}</td>`).join('')+'</tr>').join('');
document.getElementById('import-preview').style.display='block';
document.getElementById('btn-import-ok').style.display='inline-flex';
}
function splitLine(line,sep){
const res=[];let cur='';let inQ=false;
for(let i=0;i<line.length;i++){if(line[i]==='"')inQ=!inQ;else if(line[i]===sep&&!inQ){res.push(cur);cur='';}else cur+=line[i];}
res.push(cur);return res;
}

// Scoring engine
function evaluarEquipo(eq){
const reasons=[];let score=0;
const obs=(eq.OBSERVACIONES||'').toLowerCase();
const bat=(eq.BATERIA||'').toLowerCase();
const sis=(eq.SISTEMA||'').toLowerCase();
const proc=(eq.PROCESADOR||'').toLowerCase();
const mem=(eq.MEMORIA||'').toLowerCase();
const disco=(eq.ALMACENAMIENTO||'').toLowerCase();
const tipo=(eq.TIPO||'').toLowerCase();
const marca=(eq.MARCA||'').toLowerCase();

// Observaciones críticas
if(obs.includes('baja')||obs.includes('inactivo')){score+=40;reasons.push({t:'Equipo de baja',sev:'critical'});}
if(obs.includes('dañado')||obs.includes('danado')||obs.includes('falla')){score+=35;reasons.push({t:'Falla reportada',sev:'critical'});}
if(obs.includes('sin sistema')||obs.includes('sin so')){score+=30;reasons.push({t:'Sin sistema operativo',sev:'critical'});}

// Batería
if(bat.includes('cambiar')||bat.includes('mala')||bat.includes('falla')){score+=25;reasons.push({t:'Batería a reemplazar',sev:'critical'});}
else if(bat.includes('revisar')||bat.includes('baja')){score+=12;reasons.push({t:'Batería deteriorada',sev:'medium'});}

// SO obsoleto
if(sis.includes('windows 7')||sis.includes('windows xp')||sis.includes('win 7')){score+=30;reasons.push({t:'SO obsoleto (Win 7/XP)',sev:'critical'});}
else if(sis.includes('windows 8')){score+=18;reasons.push({t:'SO próximo a EOL (Win 8)',sev:'medium'});}
else if(sis.includes('windows 10')){score+=6;reasons.push({t:'Win 10 — soporte hasta oct 2025',sev:'low'});}

// Procesador antiguo
if(proc.includes('core 2')||proc.includes('pentium d')||proc.includes('celeron')||proc.includes('i3-2')||proc.includes('i5-2')||proc.includes('i5-3')||proc.includes('i3-3')){score+=20;reasons.push({t:'CPU de generación antigua',sev:'medium'});}
else if(proc.includes('i3-4')||proc.includes('i5-4')||proc.includes('i7-4')||proc.includes('i3-5')||proc.includes('i5-5')){score+=10;reasons.push({t:'CPU generación 4ª/5ª',sev:'low'});}

// Memoria baja
const memN=parseInt(mem)||0;
if(memN>0&&memN<=2){score+=25;reasons.push({t:'Memoria ≤2 GB',sev:'critical'});}
else if(memN===4){score+=12;reasons.push({t:'Memoria 4 GB',sev:'medium'});}

// Disco HDD
if(disco.includes('hdd')){score+=10;reasons.push({t:'Disco HDD (no SSD)',sev:'low'});}
if(disco.includes('32 gb')||disco.includes('64 gb')){score+=18;reasons.push({t:'Almacenamiento insuficiente',sev:'medium'});}

// Observaciones pendientes
if(obs.includes('pendiente')||obs.includes('revisar')){score+=8;reasons.push({t:'Pendiente de revisión',sev:'low'});}
if(obs.includes('sin antivirus')){score+=10;reasons.push({t:'Sin antivirus',sev:'medium'});}

let severity='ok';
if(score>=40) severity='critical';
else if(score>=20) severity='medium';
else if(score>=8) severity='low';

return{score,severity,reasons};
}

// ══════════ CRYPTO (SHA-256 via Web Crypto) ══════════
async function sha256(text){
const buf=await crypto.subtle.digest('SHA-256',new TextEncoder().encode(text));
return Array.from(new Uint8Array(buf)).map(b=>b.toString(16).padStart(2,'0')).join('');
}
